Giải thích các câu lệnh: 

git commit --amend : Sửa commit gần nhất
git commit --amend -m"<message>" : Thay đổi message của commit gần nhất
// Sau khi commit nhưng muốn sửa message
git commit -m "Lỗi"
git commit --amend -m "Sửa lỗi"


git restore --staged <file> : Hủy stage file đã được thêm vào staging area
# Thêm sai file vào staging
git add file-sai.txt
# Hủy stage file đó
git restore --staged file-sai.txt
# File thay đổi vẫn còn trong working directory
# Nhưng không còn trong staging area

git reset HEAD~1 : Hủy commit gần nhất và đưa các thay đổi trở lại staging area
# File đã commit sẽ trở lại staging area
# Bây giờ có thể sửa và commit lại
